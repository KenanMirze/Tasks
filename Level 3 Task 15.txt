#include <iostream>
using namespace std;
int main() {
    int s, e, c=0;
    cout << "Neçə ədəd daxil etmək istəyirsiniz? ";
    cin >> s;
    for (int i=1; i<=s; i++) {
        cout << i << "-ci ədədi daxil edin: ";
        cin >> e;
        c+=e;
    }
    cout << "Daxil edilən ədədlərin cəmi: " << c << endl;
    return 0;
}
