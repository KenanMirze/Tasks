#include <iostream>
using namespace std;

int main() {
    double a, b;
    char op;

    cout << "Birinci ededi daxil edin: ";
    cin >> a;

    cout << "Operatoru daxil edin (+, -, *, /): ";
    cin >> op;

    cout << "Ikinci ededi daxil edin: ";
    cin >> b;

    switch (op) {
        case '+':
            cout << "Netice: " << a + b << endl;
            break;
        case '-':
            cout << "Netice: " << a - b << endl;
            break;
        case '*':
            cout << "Netice: " << a * b << endl;
            break;
        case '/':
            if (b != 0)
                cout << "Netice: " << a / b << endl;
            else
                cout << "Xeta: 0-a bolmek olmaz!" << endl;
            break;
        default:
            cout << "Xeta: duzgun operator daxil edin!" << endl;
    }

    return 0;
}
